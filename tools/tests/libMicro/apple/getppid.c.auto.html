#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>

#include "../libmicro.h"

int
benchmark_init()
{
	(void) sprintf(lm_usage, "note: measures getppid()");

	lm_tsdsize = 0;

	return (0);
}

/*ARGSUSED*/
int
benchmark(void *tsd, result_t *res)
{
	int			i;

	for (i = 0; i < lm_optB; i ++) {
		(void) getppid();
	}
	res->re_count = i;

	return (0);
}
